#include <stdio.h>
#include <inttypes.h>
#include <math.h>
#define MAX_N 1000000

int64_t is_cublic(int64_t n) {
    int64_t low = 1, high = MAX_N, mid;
    while (low <= high) {  //当这个区间还有数字时，一直找下去
        mid = (low + high) >> 1;  //mid置为中间位置
        if (pow(mid, 3) == n) return 1;  //如果找到，则返回1
        if (pow(mid, 3) < n) low = mid + 1;  //如果中间的立方数数小于n，则在后半段找
        else high = mid - 1;  //否则在前半段找
    }
    return 0;
}

int main() {
    int64_t n;
    while (scanf("%" PRId64,&n) != EOF) {
        printf("%" PRId64 " %s cublic\n", n, is_cublic(n) ? "is" : "is not");
    }
    return 0;
}
