#### 1.欧拉第9题

```c
//法一：
#include <stdio.h>

int is_triplet(int a, int b, int c) {
    if (a * a + b * b == c * c) return 1;
    else return 0;
}

int main() {
    for (int i = 3; i < 900; i++) {
        for (int j = i + 1; j < 1000; j++) {
            if (!is_triplet(i, j, (1000 - i - j))) continue;
            printf("%d\n",i * j * (1000 - i - j));
            break;
        }
    }
    return 0;
}
```

#### 2.欧拉第20题

```c
#include <stdio.h>
#define MAX_N 10000

int num[MAX_N + 5] = {1, 1};

int sum_digit(int n) {
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= num[0]; j++) {
            num[j] *= i;
        }
        for (int j = 1; j <= num[0]; j++) {
            if (num[j] < 10) continue;
            num[j + 1] += num[j] / 10;
            num[j] = num[j] % 10;
            num[0] += (j == num[0]);
        }
    }
    int ans = 0;
    for (int i = 1; i <= num[0]; i++) {
        ans += num[i];
    }
    return ans;
}

int main() {
    int n = 100;
    printf("%d\n", sum_digit(n));
    return 0;
}
```

#### 3.欧拉第22题

```c
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <inttypes.h>
#include <algorithm>
#include "22.h"

int main() {
    int64_t i = 0, ans = 0;
    while (nameList[i].length() != 0) i++;
    std::sort(nameList, nameList + i);
    i = 0;
    while (nameList[i].length() != 0) {
        int64_t sum = 0;
        for (int j = 0; j < nameList[i].length(); j++) {
            sum += nameList[i][j] - 'A' + 1;
        }
        ans += sum * (i + 1);
        i++;
    }
    printf("%" PRId64 "\n", ans);
    return 0;
}
```
